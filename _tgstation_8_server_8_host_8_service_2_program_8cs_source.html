149 public async Task OnExecuteAsync()
150 {
151 var standardRun = !Install && !Uninstall && !Configure;
152 if (standardRun)
153 if (!Silent && !WindowsServiceHelpers.IsWindowsService())
154 {
155 var result = NativeMethods.MessageBox(
156 default,
157 "You are running the TGS windows service executable directly. It should only be run by the service control manager. Would you like to install and configure the service in this location?",
158 "TGS Service",
159 NativeMethods.MessageBoxButtons.YesNo);
160
161 if (result != NativeMethods.DialogResult.Yes)
162 return;

164 Install = true;
165 Configure = true;
166 }
167 else
168 using (var service = new ServerService(WatchdogFactory, GetPassthroughArgs(), LogLevel.Trace))

170 service.Run();

173
174 if (Configure)

177 bool stopped = false;

194 stopped = true;
195 }

197 if (Install)

200 if (Restart)

202 using (sc)
203 if (sc.ServiceName == ServerService.Name)

205 if (!stopped)

208 sc.Start();

216 bool RunServiceInstall()
217 {
218 // First check if the service already exists
219 bool serviceStopped = false;
220 if (Force || !WindowsServiceHelpers.IsWindowsService())
221 foreach (ServiceController sc in ServiceController.GetServices())
222 using (sc)
223 {
224 var serviceName = sc.ServiceName;
225 if (serviceName == ServerService.Name || serviceName == "tgstation-server-4")
226 {
227 NativeMethods.DialogResult result = !Force
228 ? NativeMethods.MessageBox(
229 default,
230 $"You already have another TGS service installed ({sc.ServiceName}). Would you like to uninstall it now? Pressing \"No\" will cancel this install.",
231 "TGS Service",
232 NativeMethods.MessageBoxButtons.YesNo)
233 : NativeMethods.DialogResult.Yes;
234 if (result != NativeMethods.DialogResult.Yes)
235 return false; // is this needed after exit?
236
237 // Stop it first to give it some cleanup time
238 RestartService(sc);
239
240 // And remove it
241 InvokeSC(sc.ServiceName);
242 }
243 }
244
245 InvokeSC(null);
246
247 return serviceStopped;

254 void RestartService(ServiceController serviceController)
255 {
256 if (serviceController.Status != ServiceControllerStatus.Running)
257 return;
258
259 var stop = !Detach;
260 if (!stop)
261 {
262 serviceController.ExecuteCommand(
263 PipeCommands.GetCommandId(
264 PipeCommands.CommandDetachingShutdown)
265 .Value);
266 serviceController.WaitForStatus(ServiceControllerStatus.Stopped, TimeSpan.FromSeconds(30));
267 if (serviceController.Status != ServiceControllerStatus.Stopped)
268 stop = true;
269 }
270
271 if (stop)
272 {
273 serviceController.Stop();
274 serviceController.WaitForStatus(ServiceControllerStatus.Stopped);
275 }
