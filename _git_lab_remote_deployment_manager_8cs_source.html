34 public override async Task<IReadOnlyCollection<TestMerge>> RemoveMergedTestMerges(
35 IRepository repository,
36 RepositorySettings repositorySettings,
37 RevisionInformation revisionInformation,
38 CancellationToken cancellationToken)
39 {
40 ArgumentNullException.ThrowIfNull(repository);
41 ArgumentNullException.ThrowIfNull(repositorySettings);
42 ArgumentNullException.ThrowIfNull(revisionInformation);
43
44 if (revisionInformation.ActiveTestMerges?.Any() != true)
45 {
46 Logger.LogTrace("No test merges to remove.");
47 return Array.Empty<TestMerge>();
48 }
49
50 var client = repositorySettings.AccessToken != null
51 ? new GitLabClient(GitLabRemoteFeatures.GitLabUrl, repositorySettings.AccessToken)
52 : new GitLabClient(GitLabRemoteFeatures.GitLabUrl);
53
54 var tasks = revisionInformation
55 .ActiveTestMerges
56 .Select(x => client
57 .MergeRequests
58 .GetAsync(
59 $"{repository.RemoteRepositoryOwner}/{repository.RemoteRepositoryName}",
60 x.TestMerge.Number)
61 .WaitAsync(cancellationToken));
62 try
63 {
64 await Task.WhenAll(tasks);
65 }
66 catch (Exception ex) when (ex is not OperationCanceledException)
67 {
68 Logger.LogWarning(ex, "Merge requests update check failed!");
69 }
70
71 var newList = revisionInformation.ActiveTestMerges.Select(x => x.TestMerge).ToList();
72
73 MergeRequest lastMerged = null;
74 async Task CheckRemoveMR(Task<MergeRequest> task)
75 {
76 var mergeRequest = await task;
77 if (mergeRequest.State != MergeRequestState.Merged)
78 return;
79
80 // We don't just assume, actually check the repo contains the merge commit.
81 if (await repository.ShaIsParent(mergeRequest.MergeCommitSha, cancellationToken))
82 {
83 if (lastMerged == null || lastMerged.ClosedAt < mergeRequest.ClosedAt)
84 lastMerged = mergeRequest;
85 newList.Remove(
86 newList.First(
87 potential => potential.Number == mergeRequest.Id));
88 }
89 }
90
91 foreach (var prTask in tasks)
92 await CheckRemoveMR(prTask);
93
94 return newList;
95 }
