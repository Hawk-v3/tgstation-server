284 protected virtual async Task<TResult> RunRequest<TResult>(
285 string route,
286 HttpContent? content,
287 HttpMethod method,
288 long? instanceId,
289 bool tokenRefresh,
290 CancellationToken cancellationToken)
291 {
292 if (route == null)
293 throw new ArgumentNullException(nameof(route));
294 if (method == null)
295 throw new ArgumentNullException(nameof(method));
296 if (content == null && (method == HttpMethod.Post || method == HttpMethod.Put))
297 throw new InvalidOperationException("content cannot be null for POST or PUT!");
298
299 HttpResponseMessage response;
300 var fullUri = new Uri(Url, route);
301 var serializerSettings = GetSerializerSettings();
302 var fileDownload = typeof(TResult) == typeof(Stream);
303 using (var request = new HttpRequestMessage(method, fullUri))
304 {
305 if (content != null)
306 request.Content = content;
307
308 try
309 {
310 var headersToUse = tokenRefresh ? tokenRefreshHeaders! : headers;
311 headersToUse.SetRequestHeaders(request.Headers, instanceId);
312
313 if (authless)
314 request.Headers.Remove(HeaderNames.Authorization);
315
316 if (fileDownload)
317 request.Headers.Accept.Add(new MediaTypeWithQualityHeaderValue(MediaTypeNames.Application.Octet));
318
319 await Task.WhenAll(requestLoggers.Select(x => x.LogRequest(request, cancellationToken))).ConfigureAwait(false);
320
321 response = await httpClient.SendAsync(request, HttpCompletionOption.ResponseHeadersRead, cancellationToken).ConfigureAwait(false);
322 }
323 finally
324 {
325 // prevent content param from getting disposed
326 request.Content = null;
327 }
328 }
329
330 try
331 {
332 await Task.WhenAll(requestLoggers.Select(x => x.LogResponse(response, cancellationToken))).ConfigureAwait(false);
333
334 // just stream
335 if (fileDownload && response.IsSuccessStatusCode)
336 return (TResult)(object)await CachedResponseStream.Create(response).ConfigureAwait(false);
337 }
338 catch
339 {
340 response.Dispose();
341 throw;
342 }
343
344 using (response)
345 {
346 var json = await response.Content.ReadAsStringAsync().ConfigureAwait(false);
347
348 if (!response.IsSuccessStatusCode)
349 {
350 if (!tokenRefresh
351 && response.StatusCode == HttpStatusCode.Unauthorized

353 return await RunRequest<TResult>(route, content, method, instanceId, false, cancellationToken).ConfigureAwait(false);

377 async Task<bool> RefreshToken(CancellationToken cancellationToken)
378 {
379 if (tokenRefreshHeaders == null)
380 return false;
381
382 var startingToken = headers.Token;
383 await semaphoreSlim.WaitAsync(cancellationToken).ConfigureAwait(false);
384 try
385 {
386 if (startingToken != headers.Token)
387 return true;
388
389 var token = await RunRequest<object, TokenResponse>(Routes.Root, new object(), HttpMethod.Post, null, true, cancellationToken).ConfigureAwait(false);
390 headers = new ApiHeaders(headers.UserAgent!, token.Bearer!);
391 }
392 catch (ClientException)
393 {
394 return false;
395 }
396 finally
397 {
398 semaphoreSlim.Release();
399 }
400
401 return true;
