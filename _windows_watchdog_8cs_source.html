142 protected override async Task HandleNewDmbAvailable(CancellationToken cancellationToken)
143 {
144 IDmbProvider compileJobProvider = DmbFactory.LockNextDmb(1);
145 bool canSeamlesslySwap = true;
146
147 if (compileJobProvider.CompileJob.ByondVersion != ActiveCompileJob.ByondVersion)
148 {
149 // have to do a graceful restart
150 Logger.LogDebug(
151 "Not swapping to new compile job {0} as it uses a different BYOND version ({1}) than what is currently active {2}. Queueing graceful restart instead...",
152 compileJobProvider.CompileJob.Id,
153 compileJobProvider.CompileJob.ByondVersion,
154 ActiveCompileJob.ByondVersion);
155 canSeamlesslySwap = false;
156 }
157
158 if (compileJobProvider.CompileJob.DmeName != ActiveCompileJob.DmeName)
159 {
160 Logger.LogDebug(
161 "Not swapping to new compile job {0} as it uses a different .dmb name ({1}) than what is currently active {2}. Queueing graceful restart instead...",
162 compileJobProvider.CompileJob.Id,
163 compileJobProvider.CompileJob.DmeName,
164 ActiveCompileJob.DmeName);
165 canSeamlesslySwap = false;
166 }
167
168 if (!canSeamlesslySwap)
169 {
170 compileJobProvider.Dispose();
171 await base.HandleNewDmbAvailable(cancellationToken);
172 return;
173 }
174
175 SwappableDmbProvider windowsProvider = null;
176 bool suspended = false;
177 try
178 {
179 windowsProvider = new SwappableDmbProvider(compileJobProvider, GameIOManager, symlinkFactory);
180
181 Logger.LogDebug("Swapping to compile job {0}...", windowsProvider.CompileJob.Id);
182 try
183 {
184 Server.Suspend();
185 suspended = true;
186 }
187 catch (Exception ex)
188 {
189 Logger.LogWarning(ex, "Exception while suspending server!");
190 }
191
192 await windowsProvider.MakeActive(cancellationToken);
193 }
194 catch (Exception ex)
195 {
196 Logger.LogError(ex, "Exception while swapping");
197 IDmbProvider providerToDispose = windowsProvider ?? compileJobProvider;
198 providerToDispose.Dispose();
199 throw;
200 }
201
202 // Let this throw hard if it fails
203 if (suspended)
204 Server.Resume();
205
206 pendingSwappable?.Dispose();
207 pendingSwappable = windowsProvider;
208 }
