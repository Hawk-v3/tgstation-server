141 protected override async ValueTask<MonitorAction> HandleNormalReboot(CancellationToken cancellationToken)
142 {
143 if (pendingSwappable != null)
144 {
145 var updateTask = BeforeApplyDmb(pendingSwappable.CompileJob, cancellationToken);
146
147 if (!pendingSwappable.Swapped)
148 {
149 // IMPORTANT: THE SESSIONCONTROLLER SHOULD STILL BE PROCESSING THE BRIDGE REQUEST SO WE KNOW DD IS SLEEPING
150 // OTHERWISE, IT COULD RETURN TO /world/Reboot() TOO EARLY AND LOAD THE WRONG .DMB
151 if (!Server.ProcessingRebootBridgeRequest)
152 {
153 // integration test logging will catch this
154 Logger.LogError(
155 "The reboot bridge request completed before the watchdog could suspend the server! This can lead to buggy DreamDaemon behaviour and should be reported! To ensure stability, we will need to hard reboot the server");
156 await updateTask;
157 return MonitorAction.Restart;
158 }
159
160 await PerformDmbSwap(pendingSwappable, cancellationToken);
161 }
162
163 var currentCompileJobId = Server.ReattachInformation.Dmb.CompileJob.Id;
164
165 await DrainDeploymentCleanupTasks(false);
166
167 IDisposable lingeringDeployment;
168 var localDeploymentCleanupGate = new TaskCompletionSource();
169 async Task CleanupLingeringDeployment()
170 {
171 var lingeringDeploymentExpirySeconds = ActiveLaunchParameters.StartupTimeout.Value;
172 Logger.LogDebug(
173 "Holding old deployment {compileJobId} for up to {expiry} seconds...",
174 currentCompileJobId,
175 lingeringDeploymentExpirySeconds);
176
177 var timeout = AsyncDelayer.Delay(TimeSpan.FromSeconds(lingeringDeploymentExpirySeconds), cancellationToken);
178
179 var completedTask = await Task.WhenAny(
180 localDeploymentCleanupGate.Task,
181 timeout);
182
183 var timedOut = completedTask == timeout;
184 Logger.Log(
185 timedOut
186 ? LogLevel.Warning
187 : LogLevel.Trace,
188 "Releasing old deployment {compileJobId}{afterTimeout}",
189 currentCompileJobId,
190 timedOut
191 ? " due to timeout!"
192 : "...");
193
194 lingeringDeployment.Dispose();
195 }
196
197 var oldDeploymentCleanupGate = Interlocked.Exchange(ref deploymentCleanupGate, localDeploymentCleanupGate);
198 oldDeploymentCleanupGate?.TrySetResult();
199
200 Logger.LogTrace("Replacing activeSwappable with pendingSwappable...");
201
202 lock (deploymentCleanupTasks)
203 {
204 lingeringDeployment = Server.ReplaceDmbProvider(pendingSwappable);
205 deploymentCleanupTasks.Add(
206 CleanupLingeringDeployment());
207 }
208
209 ActiveSwappable = pendingSwappable;
210 pendingSwappable = null;
211
212 await SessionPersistor.Save(Server.ReattachInformation, cancellationToken);
213 await updateTask;
214 }
215 else
216 Logger.LogTrace("Nothing to do as pendingSwappable is null.");
217
218 return MonitorAction.Continue;
219 }

222 protected override async ValueTask HandleNewDmbAvailable(CancellationToken cancellationToken)
223 {
224 IDmbProvider compileJobProvider = DmbFactory.LockNextDmb(1);
225 bool canSeamlesslySwap = true;
226
227 if (compileJobProvider.CompileJob.ByondVersion != ActiveCompileJob.ByondVersion)
228 {
229 // have to do a graceful restart
230 Logger.LogDebug(
231 "Not swapping to new compile job {0} as it uses a different BYOND version ({1}) than what is currently active {2}. Queueing graceful restart instead...",
232 compileJobProvider.CompileJob.Id,
233 compileJobProvider.CompileJob.ByondVersion,
234 ActiveCompileJob.ByondVersion);
235 canSeamlesslySwap = false;
236 }
237
238 if (compileJobProvider.CompileJob.DmeName != ActiveCompileJob.DmeName)
239 {
240 Logger.LogDebug(
241 "Not swapping to new compile job {0} as it uses a different .dmb name ({1}) than what is currently active {2}. Queueing graceful restart instead...",
242 compileJobProvider.CompileJob.Id,
243 compileJobProvider.CompileJob.DmeName,
244 ActiveCompileJob.DmeName);
245 canSeamlesslySwap = false;
246 }
247
248 if (!canSeamlesslySwap)
249 {
250 compileJobProvider.Dispose();
251 await base.HandleNewDmbAvailable(cancellationToken);
252 return;
253 }
254
255 SwappableDmbProvider windowsProvider = null;
256 try
257 {
258 windowsProvider = new SwappableDmbProvider(compileJobProvider, GameIOManager, symlinkFactory);
259 if (ActiveCompileJob.DMApiVersion == null)
260 {
261 Logger.LogWarning("Active compile job has no DMAPI! Commencing immediate .dmb swap. Note this behavior is known to be buggy in some DM code contexts. See https://github.com/tgstation/tgstation-server/issues/1550");
262 await PerformDmbSwap(windowsProvider, cancellationToken);
263 }
264 }
265 catch (Exception ex)
266 {
267 Logger.LogError(ex, "Exception while swapping");
268 IDmbProvider providerToDispose = windowsProvider ?? compileJobProvider;
269 providerToDispose.Dispose();
270 throw;
271 }
272
273 pendingSwappable?.Dispose();
274 pendingSwappable = windowsProvider;
275 }

347 async ValueTask PerformDmbSwap(SwappableDmbProvider newProvider, CancellationToken cancellationToken)
348 {
349 Logger.LogDebug("Swapping to compile job {id}...", newProvider.CompileJob.Id);
350
351 var suspended = false;
352 var server = Server;
353 try
354 {
355 server.Suspend();
356 suspended = true;
357 }
358 catch (Exception ex)
359 {
360 Logger.LogWarning(ex, "Exception while suspending server!");
361 }
362
363 try
364 {
365 await newProvider.MakeActive(cancellationToken);
366 }
367 finally
368 {
369 // Let this throw hard if it fails
370 if (suspended)
371 server.Resume();
372 }
373 }

380 Task DrainDeploymentCleanupTasks(bool blocking)
381 {
382 Logger.LogTrace("DrainDeploymentCleanupTasks...");
383 var localDeploymentCleanupGate = Interlocked.Exchange(ref deploymentCleanupGate, null);
384 localDeploymentCleanupGate?.TrySetResult();
385
386 List<Task> localDeploymentCleanupTasks;
387 lock (deploymentCleanupTasks)
388 {
389 var totalActiveTasks = deploymentCleanupTasks.Count;
390 localDeploymentCleanupTasks = new List<Task>(totalActiveTasks);
391 for (var i = totalActiveTasks - 1; i >= 0; --i)
392 {
393 var currentTask = deploymentCleanupTasks[i];
394 if (!blocking && !currentTask.IsCompleted)
395 continue;
396
397 localDeploymentCleanupTasks.Add(currentTask);
398 deploymentCleanupTasks.RemoveAt(i);
399 }
400 }
401
402 return Task.WhenAll(localDeploymentCleanupTasks);
403 }
