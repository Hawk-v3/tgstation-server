229 async Task AddDreamDaemonToFirewall(Version version, string path, CancellationToken cancellationToken)
230 {
231 var dreamDaemonName = GetDreamDaemonName(version, out var usesDDExe, out var _);
232
233 var dreamDaemonPath = IOManager.ResolvePath(
234 IOManager.ConcatPath(
235 path,
236 ByondManager.BinPath,
237 dreamDaemonName));
238
239 Logger.LogInformation("Adding Windows Firewall exception for {path}...", dreamDaemonPath);
240 try
241 {
242 // I really wish we could add the instance name here but
243 // 1. It'd make IByondInstaller need to be transient per-instance and WindowsByondInstaller relys on being a singleton for its DX installer call
244 // 2. The instance could be renamed, so it'd have to be an unfriendly ID anyway.
245 var arguments = $"advfirewall firewall add rule name=\"TGS DreamDaemon {version}\" program=\"{dreamDaemonPath}\" protocol=tcp dir=in enable=yes action=allow";
246 await using var netshProcess = processExecutor.LaunchProcess(
247 "netsh.exe",
248 IOManager.ResolvePath(),
249 arguments,
250 readStandardHandles: true,
251 noShellExecute: true);
252
253 int exitCode;
254 using (cancellationToken.Register(() => netshProcess.Terminate()))
255 exitCode = (await netshProcess.Lifetime).Value;
256 cancellationToken.ThrowIfCancellationRequested();
257
258 Logger.LogDebug(
259 "netsh.exe output:{newLine}{output}",
260 Environment.NewLine,
261 await netshProcess.GetCombinedOutput(cancellationToken));
262
263 if (exitCode != 0)
264 throw new JobException(ErrorCode.ByondDreamDaemonFirewallFail, new JobException($"Invalid exit code: {exitCode}"));
265
266 if (usesDDExe)
267 await IOManager.WriteAllBytes(
268 IOManager.ConcatPath(path, TgsFirewalledDDFile),
269 Array.Empty<byte>(),
270 cancellationToken);
271 }
272 catch (Exception ex)
273 {
274 throw new JobException(ErrorCode.ByondDreamDaemonFirewallFail, ex);
