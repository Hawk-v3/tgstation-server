141 public static IServiceCollection UseStandardConfig<TConfig>(this IServiceCollection serviceCollection, IConfiguration configuration) where TConfig : class
142 {
143 ArgumentNullException.ThrowIfNull(serviceCollection);
144 ArgumentNullException.ThrowIfNull(configuration);
145
146 const string SectionFieldName = nameof(GeneralConfiguration.Section);
147
148 var configType = typeof(TConfig);
149 var sectionField = configType.GetField(SectionFieldName) ?? throw new InvalidOperationException(
150 String.Format(CultureInfo.InvariantCulture, "{0} has no {1} field!", configType, SectionFieldName));
151 var stringType = typeof(string);
152 if (sectionField.FieldType != stringType)
153 throw new InvalidOperationException(String.Format(CultureInfo.InvariantCulture, "{0} has invalid {1} field type, must be {2}!", configType, SectionFieldName, stringType));
154
155 var sectionName = (string)sectionField.GetValue(null);
156
157 return serviceCollection.Configure<TConfig>(configuration.GetSection(sectionName));
158 }

170 public static IServiceCollection SetupLogging(
171 this IServiceCollection serviceCollection,
172 Action<LoggerConfiguration> configurationAction,
173 Action<LoggerSinkConfiguration> sinkConfigurationAction = null,
174 ElasticsearchSinkOptions elasticsearchSinkOptions = null,
175 InternalConfiguration internalConfiguration = null,
176 FileLoggingConfiguration fileLoggingConfiguration = null)
177 {
178 if (internalConfiguration != null)
179 ArgumentNullException.ThrowIfNull(fileLoggingConfiguration);
180
181 return serviceCollection.AddLogging(builder =>
182 {
183 builder.ClearProviders();
184
185 var configuration = new LoggerConfiguration()
186 .MinimumLevel
187 .Verbose();
188
189 configurationAction?.Invoke(configuration);
190
191 configuration
192 .Enrich.FromLogContext()
193 .WriteTo
194 .Async(sinkConfiguration =>
195 {
196 var template = "[{Timestamp:HH:mm:ss}] {Level:w3}: {SourceContext:l} ("
197 + SerilogContextHelper.Template
198 + "){NewLine} {Message:lj}{NewLine}{Exception}";
199
200 if (!((internalConfiguration?.UsingSystemD ?? false) && !fileLoggingConfiguration.Disable))
201 sinkConfiguration.Console(outputTemplate: template, formatProvider: CultureInfo.InvariantCulture);
202 sinkConfigurationAction?.Invoke(sinkConfiguration);
203 });
204
205 if (elasticsearchSinkOptions != null)
206 configuration.WriteTo.Elasticsearch(elasticsearchSinkOptions);
207
208 builder.AddSerilog(configuration.CreateLogger(), true);
209
210 if (Debugger.IsAttached)
211 builder.AddDebug();
212
213 if (additionalLoggerProvider != null)
214 builder.Services.TryAddEnumerable(additionalLoggerProvider);
